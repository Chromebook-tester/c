  <h1>🔊 スピーカーテスト</h1>

  <button onclick="toggleAudio('left', this)">1. 左 スピーカー</button>
  <button onclick="toggleAudio('right', this)">2. 右 スピーカー</button>
  <br>
  <button onclick="toggleAudio('center', this)">3. 左右 スピーカー</button>

  <audio id="left" src="left-speaker.mp3"></audio>
  <audio id="right" src="right-speaker.mp3"></audio>
  <audio id="center" src="center-speaker.mp3"></audio>

  <script>
    let currentAudio = null;
    let currentButton = null;

    function toggleAudio(id, button) {
      const audio = document.getElementById(id);

      // 같은 버튼 다시 누름 → 정지
      if (currentAudio === audio && !audio.paused) {
        audio.pause();
        audio.currentTime = 0;
        currentAudio = null;
        currentButton.classList.remove("active");
        currentButton = null;
        return;
      }

      // 기존 오디오 정지
      if (currentAudio) {
        currentAudio.pause();
        currentAudio.currentTime = 0;
      }
      if (currentButton) {
        currentButton.classList.remove("active");
      }

      // 새로 재생
      currentAudio = audio;
      currentButton = button;

      audio.play().then(() => {
        button.classList.add("active");
      }).catch(err => {
        alert("再生できませんでした: " + err);
      });

      // ✅ 버튼 색상 유지: 끝나도 active 클래스 제거하지 않음
    }
  </script>
