  <script>
    function playAudio(id) {
      const sounds = ['left', 'right', 'center', 'home', 'next', 'prev'];
      sounds.forEach(s => {
        const audio = document.getElementById(s);
        audio.pause();
        audio.currentTime = 0;
      });

      const target = document.getElementById(id);
      if (target) {
        target.play().catch(err => {
          alert("再生できませんでした: " + err.message);
        });
      }
    }
  </script>
